Функция линейная_регрессия(X, y):
    
    # Находим количество примеров
    m = количество_строк(X)
    
    # Инициализируем веса
    theta_0 = 0
    theta_1 = 0
    
    # Вычисляем ошибку
    ошибка = бесконечность
    
    # Пока ошибка больше заданного порога
    пока ошибка > порог:
        
        # Вычисляем прогнозы
        y_pred = theta_0 + theta_1 * X
        
        # Вычисляем квадратичную ошибку
        ошибка = сумма((y_pred - y) ** 2) / (2 * m)
        
        # Обновляем веса
        theta_0 = theta_0 - скорость_обучения * (1/m) * сумма(y_pred - y)
        theta_1 = theta_1 - скорость_обучения * (1/m) * сумма((y_pred - y) * X)
    
    # Возвращаем найденные веса
    return theta_0, theta_1
